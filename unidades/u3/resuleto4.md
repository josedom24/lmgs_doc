		while palabra != " ":
			lista.append(palabra)
			palabra=input("Palabra:")

		sustituir=input("Palabra a sustituir:")		

		cont=0
		for cad in lista:
			if cad==buscar:
				lista[cont]=sustituir
			cont=cont+1		

		print(lista)
